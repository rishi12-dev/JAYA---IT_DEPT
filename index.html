// Welcome to Jaya IT Department
class Student {
    constructor(name, year) {
        this.name = name;
        this.year = year;
        this.skills = [];
    }

    addSkill(skill) {
        this.skills.push(skill);
        return this;
    }

    graduate() {
        return `Congratulations ${this.name}!`;
    }
}

const itStudent = new Student('Future Engineer', 2023);

itStudent
    .addSkill('JavaScript')
    .addSkill('Python')
    .addSkill('Machine Learning');

console.log(itStudent.graduate());
